**前端调用示例**:
```typescript
// Gojica2.0前端/common/utils/api.uts
export const authApi = {
  devLogin: (phone: string, code: string, identity?: string) => 
    apiPost('/auth/dev-login', { phone, code, identity }),
  login: (code: string, identity?: string) => 
    apiPost('/auth/login', { code, identity }),
  logout: () => apiPost('/auth/logout'),
  refresh: (refreshToken: string) => 
    apiPost('/auth/refresh', { refreshToken }),
  sendCode: (phone: string) => 
    apiPost('/auth/send-code', { phone })
}
```

### 4.1 登录 → 首页 → 乐队 → 活动完整流程

```typescript
describe('完整业务流程 E2E', () => {
  it('登录 → 浏览 → 关注 → 报名', async () => {
    // 1. 登录
    await loginAsFan()
    
    // 2. 进入首页
    await page.goto('/pages/home/index')
    await expect(page.locator('.stats-section')).toBeVisible()
    
    // 3. 进入乐队列表
    await page.click('[data-testid="tab-bands"]')
    await expect(page.locator('.band-list')).toBeVisible()
    
    // 4. 进入乐队详情
    const firstBand = page.locator('.band-item').first()
    await firstBand.click()
    await expect(page.locator('.band-detail')).toBeVisible()
    
    // 5. 关注乐队
    const followBtn = page.locator('button:has-text("关注")')
    await followBtn.click()
    await expect(page.locator('button:has-text("取消关注")')).toBeVisible()
    
    // 6. 返回并进入活动列表
    await page.click('.back-btn')
    await page.click('[data-testid="tab-activities"]')
    await expect(page.locator('.activity-list')).toBeVisible()
    
    // 7. 进入活动详情
    const firstActivity = page.locator('.activity-item').first()
    await firstActivity.click()
    await expect(page.locator('.activity-detail')).toBeVisible()
    
    // 8. 报名活动
    const signupBtn = page.locator('button:has-text("立即报名")')
    await signupBtn.click()
    await expect(page.locator('text=报名成功')).toBeVisible()
    
    // 9. 进入用户中心
    await page.click('.tab-user')
    await expect(page.locator('.user-profile')).toBeVisible()
    
    // 10. 验证我的活动中有刚才报名的活动
    await page.click('text=我的活动')
    const activity = page.locator('.activity-item:has-text("测试活动")')
    await expect(activity).toBeVisible()
    
    // 11. 退出登录
    await page.click('text=退出登录')
    await expect(page.locator('.login-page')).toBeVisible()
  })
})
```

  u.id,
  u.nickname,

  COUNT(DISTINCT a.id) as activity_count,
  COUNT(DISTINCT p.id) as post_count
FROM users u
LEFT JOIN bands b ON b.owner_id = u.id
LEFT JOIN activities a ON a.organizer_id = u.id
LEFT JOIN posts p ON p.user_id = u.id
GROUP BY u.id
LIMIT 10;

-- 验证乐队成员关系
SELECT 
  bm.band_id,
  bm.user_id,
  b.name as band_name,
  u.nickname as user_name,
  bm.role
FROM band_members bm
JOIN bands b ON b.id = bm.band_id
JOIN users u ON u.id = bm.user_id
WHERE bm.band_id IN (1, 2, 3)
ORDER BY bm.band_id, bm.created_at;

-- 验证活动报名关系
SELECT 
  a.title,
  a.max_participants,
  COUNT(s.id) as signup_count,
  GROUP_CONCAT(u.nickname) as participants
FROM activities a
LEFT JOIN activity_signups s ON s.activity_id = a.id AND s.status = 'confirmed'
LEFT JOIN users u ON u.id = s.user_id
GROUP BY a.id
HAVING signup_count > 0
LIMIT 10;

-- 验证动态点赞关系
SELECT 
  p.id,
  p.content,
  COUNT(l.id) as like_count,
  COUNT(DISTINCT l.user_id) as liker_count
FROM posts p
LEFT JOIN likes l ON l.target_id = p.id AND l.target_type = 'post'
GROUP BY p.id
ORDER BY like_count DESC
LIMIT 10;

const { performance } = require('perf_hooks');

const BASE_URL = 'http://localhost:3000/api/v1';

function httpRequest(options, postData = null) {
  return new Promise((resolve, reject) => {
    const start = performance.now();
    const req = http.request(options, (res) => {
      let data = '';
      res.on('data', (chunk) => { data += chunk; });
      res.on('end', () => {
        const duration = performance.now() - start;
        try {
          resolve({ status: res.statusCode, data: JSON.parse(data), duration });
        } catch (e) {
          resolve({ status: res.statusCode, data: data, duration });
        }
      });
    });
    req.on('error', reject);
    if (postData) req.write(postData);
    req.end();
  });
}

async function testPerformance() {
  console.log('⚡ 性能基准测试\n');
  
  const endpoints = [
    { name: 'Home', path: '/home', method: 'GET' },
    { name: 'Bands List', path: '/bands?page=1&pageSize=10', method: 'GET' },
    { name: 'Activities List', path: '/activities?page=1&pageSize=10', method: 'GET' },
    { name: 'Posts List', path: '/posts?page=1&pageSize=10', method: 'GET' },
    { name: 'Search', path: '/search?q=rock', method: 'GET' }
  ];
  
  const iterations = 10;
  const results = [];
  
  for (const endpoint of endpoints) {
    const timings = [];
    
    for (let i = 0; i < iterations; i++) {
      const res = await httpRequest({
        hostname: 'localhost', port: 3000,
        path: `/api/v1${endpoint.path}`, method: endpoint.method
      });
      timings.push(res.duration);
    }
    
    timings.sort((a, b) => a - b);
    const avg = timings.reduce((a, b) => a + b, 0) / timings.length;
    const p50 = timings[Math.floor(timings.length * 0.5)];
    const p95 = timings[Math.floor(timings.length * 0.95)];
    const p99 = timings[Math.floor(timings.length * 0.99)];
    
    results.push({
      name: endpoint.name,
      avg: avg.toFixed(2),
      p50: p50.toFixed(2),
      p95: p95.toFixed(2),
      p99: p99.toFixed(2)
    });
    
    console.log(`${endpoint.name}:`);
    console.log(`  平均: ${avg.toFixed(2)}ms`);
    console.log(`  P50:  ${p50.toFixed(2)}ms`);
    console.log(`  P95:  ${p95.toFixed(2)}ms`);
    console.log(`  P99:  ${p99.toFixed(2)}ms\n`);
  }
  
  console.log('✅ 性能测试完成\n');
}

testPerformance().catch(console.error);
