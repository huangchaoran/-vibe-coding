## 🎫 Token 设计

### Access Token
```javascript
{
  "header": {
    "alg": "HS256",
    "typ": "JWT"
  },
  "payload": {
    "userId": 1,
    "openid": "oXXXXXX",
    "identity": "musician",
    "role": "user",
    "iat": 1714214400,
    "exp": 1714293600  // 24小时后过期
  },
  "signature": "xxxxxx"
}
```

### Token 配置
```javascript
// server/src/config/jwt.js
module.exports = {
  secret: process.env.JWT_SECRET || 'gojica_secret_key_2024',
  expiresIn: '24h',           // Access Token 有效期
  refreshExpiresIn: '7d',     // Refresh Token 有效期
  algorithm: 'HS256',
  issuer: 'gojica-api'
};
```

### Token 生成
```javascript
// server/src/utils/jwt.js
const jwt = require('jsonwebtoken');
const config = require('../config/jwt');

const generateToken = (user) => {
  return jwt.sign(
    {
      userId: user.id,
      openid: user.openid,
      identity: user.identity,
      role: user.role || 'user'
    },
    config.secret,
    {
      expiresIn: config.expiresIn,
      issuer: config.issuer
    }
  );
};

const verifyToken = (token) => {
  return jwt.verify(token, config.secret);
};

module.exports = {
  generateToken,
  verifyToken
};
```

## 🛡️ 权限中间件

### 基础认证中间件
```javascript
// server/src/middlewares/auth.js
const { verifyToken } = require('../utils/jwt');
const { UserRole } = require('../constants');

// 需要登录
const requireAuth = (req, res, next) => {
  try {
    const token = req.headers.authorization?.replace('Bearer ', '');
    
    if (!token) {
      return res.status(401).json({
        code: 2005,
        message: '请先登录'
      });
    }
    
    const decoded = verifyToken(token);
    req.user = decoded;
    next();
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({
        code: 2004,
        message: '登录已过期，请重新登录'
      });
    }
    
    return res.status(401).json({
      code: 2003,
      message: 'Token无效'
    });
  }
};

// 需要特定身份
const requireIdentity = (...identities) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        code: 2005,
        message: '请先登录'
      });
    }
    
    if (!identities.includes(req.user.identity)) {
      return res.status(403).json({
        code: 2006,
        message: '您的身份不允许访问此功能'
      });
    }
    
    next();
  };
};

// 需要特定角色
const requireRole = (...roles) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        code: 2005,
        message: '请先登录'
      });
    }
    
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({
        code: 2006,
        message: '权限不足'
      });
    }
    
    next();
  };
};

// 需要是资源所有者或管理员
const requireOwnerOrAdmin = (getResourceUserId) => {
  return async (req, res, next) => {
    try {
      const resourceUserId = await getResourceUserId(req);
      
      if (req.user.userId === resourceUserId || 
          req.user.role === UserRole.ADMIN || 
          req.user.role === UserRole.SUPER_ADMIN) {
        return next();
      }
      
      return res.status(403).json({
        code: 2006,
        message: '权限不足，只能操作自己的资源'
      });
    } catch (error) {
      return res.status(500).json({
        code: 500,
        message: '服务器错误'
      });
    }
  };
};

module.exports = {
  requireAuth,
  requireIdentity,
  requireRole,
  requireOwnerOrAdmin
};
```

### 权限使用示例
```javascript
// server/src/routes/activities.js
const express = require('express');
const router = express.Router();
const activityController = require('../controllers/activityController');
const { requireAuth, requireIdentity, requireOwnerOrAdmin } = require('../middlewares/auth');
const { UserIdentity } = require('../constants');

// 公开接口
router.get('/', activityController.getList);
router.get('/:id', activityController.getDetail);

// 需要登录
router.post('/:id/signup', requireAuth(), activityController.signup);
router.post('/:id/cancel', requireAuth(), activityController.cancelSignup);
router.post('/:id/collect', requireAuth(), activityController.collect);

// 需要乐手或乐队身份
router.post('/', requireAuth(), requireIdentity(UserIdentity.MUSICIAN, UserIdentity.BAND), activityController.create);
router.put('/:id', requireAuth(), requireOwnerOrAdmin(async (req) => {
  const activity = await Activity.findById(req.params.id);
  return activity.organizer_id;
}), activityController.update);

module.exports = router;
```

### 登录逻辑
```javascript
// server/src/services/authService.js
const wxLogin = async (code) => {
  // 1. 调用微信接口获取 openid
  const wxResult = await wechatService.code2Session(code);
  const { openid, unionid } = wxResult;
  
  // 2. 查询用户是否已存在
  let user = await User.findOne({ where: { openid } });
  
  // 3. 如果不存在，创建新用户
  let isNewUser = false;
  if (!user) {
    isNewUser = true;
    user = await User.create({
      openid,
      unionid,
      nickname: '新用户',
      avatar: '/static/images/default-avatar.png',
      identity: 'fan'
    });
  }
  
  // 4. 更新最后登录时间
  await User.update(
    { last_login_at: new Date() },
    { where: { id: user.id } }
  );
  
  // 5. 生成 Token
  const token = jwtUtils.generateToken(user);
  
  return {
    token,
    userInfo: {
      id: user.id,
      openid: user.openid,
      nickname: user.nickname,
      avatar: user.avatar,
      identity: user.identity,
      isNewUser
    }
  };
};
```

```javascript
app.use((req, res, next) => {
  const isPublicRoute = publicRoutes.some(route => {
    const pattern = new RegExp('^' + route.replace(/:[\w]+/g, '[^/]+') + '$');
    return pattern.test(req.path);

      code: 2005,
      message: '请先登录'
```

## 🛡️ 安全策略

### 密码加密
```javascript
// 使用 bcrypt
const bcrypt = require('bcryptjs');
const SALT_ROUNDS = 10;

// 密码加密
const hashPassword = async (password) => {
  return await bcrypt.hash(password, SALT_ROUNDS);
};

// 密码验证
const comparePassword = async (password, hash) => {
  return await bcrypt.compare(password, hash);
};
```

### Token 安全
1. **Token 存储**：仅存储在客户端内存或 HttpOnly Cookie
2. **传输安全**：HTTPS 传输
3. **有效期**：Access Token 24小时，Refresh Token 7天
4. **Token 刷新**：使用 Refresh Token 获取新的 Access Token

```javascript
const trackLoginFail = (ip) => {
```

## 🔄 Token 刷新机制

### Refresh Token
```javascript
// POST /api/v1/auth/refresh
// Request Body:
{
  "refreshToken": "xxx"
}

// Response:
{
  "code": 200,
  "data": {
    "token": "new_access_token",
    "expiresIn": 86400
  }
}
```

### 实现逻辑
```javascript
const refreshToken = async (refreshToken) => {
  try {
    // 验证 Refresh Token
    const decoded = jwtUtils.verifyRefreshToken(refreshToken);
    
    // 获取用户信息
    const user = await User.findById(decoded.userId);
    
    if (!user) {
      throw new Error('用户不存在');
    }
    
    // 生成新的 Access Token
    const newToken = jwtUtils.generateToken(user);
    
    return {
      token: newToken,
      expiresIn: 86400
    };
  } catch (error) {
    throw new Error('Refresh Token 无效或已过期');
  }
};
```
